<script>

let cart = [];

 

function addToCart(name, price) {

  let item = cart.find(i => i.name === name);

  if (item) {

    item.qty++;

  } else {

    cart.push({name, price, qty:1});

  }

  updateCart();

}

 

function updateCart() {

  let total = 0;

  let count = 0;

 

  cart.forEach(i => {

    total += i.price * i.qty;

    count += i.qty;

  });

 

  document.getElementById("cartText").innerText =

    `Cart: ${count} items | ₹${total}`;

}

 

function sendToWhatsApp() {

  if(cart.length === 0){

    alert("Cart is empty!");

    return;

  }

 

  let message = "Hello, I want to order:\n\n";

 

  cart.forEach(i => {

    message += `${i.name} x ${i.qty} = ₹${i.price * i.qty}\n`;

  });

 

  let total = cart.reduce((sum, i) => sum + i.price*i.qty, 0);

  message += `\nTotal: ₹${total}`;

 

  let url = "https://wa.me/917337775777?text=" + encodeURIComponent(message);

  window.open(url, "_blank");

}

</script>
